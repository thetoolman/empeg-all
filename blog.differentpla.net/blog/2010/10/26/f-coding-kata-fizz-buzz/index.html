let IsDivisible n d =
    ((n % d) = 0)

let FizzBuzz n =
    match n with
    | n when IsDivisible n 3 && IsDivisible n 5 -> "fizzbuzz"
    | n when IsDivisible n 3 -> "fizz"
    | n when IsDivisible n 5 -> "buzz"
    | _ -> n.ToString()
